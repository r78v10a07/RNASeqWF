pvalues = []
index = 0
genes = {}
for i, r in tpm_df.iterrows():
    x = r[classical]
    y = r[nonclassical]    
    try:
        t, p = stats.mannwhitneyu(x, y,alternative='two-sided')
        pvalues.append(p)
        genes[index] = r
        genes[index]['P-Value'] = p
        genes[index]['t'] = t
        tpm1 = np.median(x)
        tpm2 = np.median(y)
        genes[index]['log2TPMRatio'] = math.log2(tpm1 / tpm2)
        index += 1
    except:
        pass

x_intron = []
y_intron = []
c_intron = []
genes_down = {}
genes_up = {}
rows = []
for index, row in mwu_df.iterrows(): 
    y_intron.append(-1.0 * math.log10(row['P-Value']))
    x_intron.append(row['log2TPMRatio'])
    if abs(row['log2TPMRatio']) >= fc and row['P-Value'] <= min_p_value:
        rows.append(row)
        if row['log2TPMRatio'] >= fc:
            c_intron.append('red')
            if row['Gene_Id'] not in genes_up:
                genes_up[row['Gene_Id']] = []
            genes_up[row['Gene_Id']].append(row)
        elif row['log2TPMRatio'] <= -fc:
            c_intron.append('blue')
            if row['Gene_Id'] not in genes_down:
                genes_down[row['Gene_Id']] = []
            genes_down[row['Gene_Id']].append(row)
    else:
        c_intron.append('lightgray')
